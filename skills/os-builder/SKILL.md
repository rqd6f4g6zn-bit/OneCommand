# OS Builder Skill

You are an expert Linux systems engineer. Your job is to generate a complete, production-quality custom Linux operating system project from a spec file. Every file you generate must be fully functional — no placeholders, no stubs.

---

## Step 1: Read Spec and Determine OS Type

Read `.onecommand-spec.json` from the current directory. Extract:

- `project_name` — used as the root directory name
- `os_type` — `"server"` | `"embedded"` | `"desktop"` | `"container"`
- `os_base` — `"alpine"` | `"buildroot"` | `"debian"` | `"arch"`
- `os_features` — array e.g. `["nginx", "postgresql", "docker", "ssh", "firewall", "monitoring"]`
- `os_hostname` — custom hostname string
- `os_packages` — explicit extra package list
- `os_services` — services to enable at boot

**Defaults when fields are missing:**
- `os_type: "server"` → `os_base: "alpine"`
- `os_type: "embedded"` → `os_base: "buildroot"`
- `os_type: "desktop"` → `os_base: "debian"`
- `os_type: "container"` → `os_base: "alpine"`

If `.onecommand-spec.json` does not exist, prompt the user for `project_name`, `os_type`, and `os_features`, then proceed with defaults.

## Step 3: Generate `build.sh` (Main Orchestration)

Write `<project_name>/build.sh`:

```bash
#!/bin/bash
# =============================================================================
# build.sh — Main OS build orchestration script
# Generated by OneCommand os-builder
# =============================================================================
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
OS_TYPE="${OS_TYPE:-server}"
OS_BASE="${OS_BASE:-alpine}"
OS_HOSTNAME="${OS_HOSTNAME:-custom-os}"
BUILD_DIR="${SCRIPT_DIR}/output"
LOG_FILE="${BUILD_DIR}/build.log"

RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m'

log()  { echo -e "${BLUE}[$(date '+%H:%M:%S')]${NC} $*" | tee -a "$LOG_FILE"; }
ok()   { echo -e "${GREEN}[OK]${NC} $*" | tee -a "$LOG_FILE"; }
warn() { echo -e "${YELLOW}[WARN]${NC} $*" | tee -a "$LOG_FILE"; }
fail() { echo -e "${RED}[FAIL]${NC} $*" | tee -a "$LOG_FILE"; exit 1; }

check_deps() {
    log "Checking build dependencies..."
    local missing=()
    for cmd in docker curl mkisofs xorriso; do
        command -v "$cmd" &>/dev/null || missing+=("$cmd")
    done
    if [[ ${#missing[@]} -gt 0 ]]; then
        warn "Missing optional tools: ${missing[*]}"
        warn "Some build targets may not be available."
    fi
    ok "Dependency check complete"
}

build_main() {
    mkdir -p "$BUILD_DIR"
    log "Starting build: OS_TYPE=${OS_TYPE}, OS_BASE=${OS_BASE}, HOSTNAME=${OS_HOSTNAME}"

    case "$OS_BASE" in
        alpine)    bash "${SCRIPT_DIR}/scripts/build-alpine.sh" ;;
        buildroot) bash "${SCRIPT_DIR}/scripts/build-buildroot.sh" ;;
        debian)    bash "${SCRIPT_DIR}/scripts/build-debian.sh" ;;
        *)         fail "Unknown OS_BASE: ${OS_BASE}" ;;
    esac

    ok "Base OS build complete"
    log "Applying overlay files..."
    if [[ -d "${SCRIPT_DIR}/overlay" ]]; then
        cp -r "${SCRIPT_DIR}/overlay/." "${BUILD_DIR}/rootfs/" 2>/dev/null || true
        ok "Overlay applied"
    fi

    log "Installing custom CLI tools..."
    for tool in os-status os-update os-backup os-monitor; do
        if [[ -f "${SCRIPT_DIR}/overlay/usr/local/bin/${tool}" ]]; then
            chmod +x "${SCRIPT_DIR}/overlay/usr/local/bin/${tool}"
            ok "Tool ready: ${tool}"
        fi
    done
}

print_summary() {
    echo ""
    echo -e "${GREEN}============================================${NC}"
    echo -e "${GREEN}  BUILD COMPLETE${NC}"
    echo -e "${GREEN}============================================${NC}"
    echo -e "  OS Type  : ${OS_TYPE}"
    echo -e "  OS Base  : ${OS_BASE}"
    echo -e "  Hostname : ${OS_HOSTNAME}"
    echo -e "  Output   : ${BUILD_DIR}/"
    echo ""
    echo -e "  Next steps:"
    echo -e "  ${YELLOW}make iso${NC}      — Create bootable ISO"
    echo -e "  ${YELLOW}make test${NC}     — Test in QEMU"
    echo -e "  ${YELLOW}make docker-test${NC} — Test in Docker"
    echo -e "${GREEN}============================================${NC}"
}

check_deps
build_main
print_summary
```

## Step 4: Generate `Makefile`

Write `<project_name>/Makefile`:

```makefile
# =============================================================================
# Makefile — OneCommand OS Builder
# =============================================================================
SHELL        := /bin/bash
PROJECT      := $(shell basename $(CURDIR))
OS_TYPE      ?= server
OS_BASE      ?= alpine
OS_HOSTNAME  ?= custom-os
OUTPUT_DIR   := output
ISO_NAME     := $(OUTPUT_DIR)/$(PROJECT).iso

.PHONY: all build iso test docker-build docker-test clean help

all: build

## build: Build the OS image
build:
	@echo "==> Building OS (base=$(OS_BASE), type=$(OS_TYPE))..."
	@OS_TYPE=$(OS_TYPE) OS_BASE=$(OS_BASE) OS_HOSTNAME=$(OS_HOSTNAME) bash build.sh

## iso: Create a bootable ISO
iso: build
	@echo "==> Creating bootable ISO..."
	@bash scripts/build-iso.sh
	@echo "ISO ready: $(ISO_NAME)"

## test: Boot OS in QEMU (requires qemu-system-x86_64)
test:
	@command -v qemu-system-x86_64 >/dev/null 2>&1 || \
		{ echo "ERROR: qemu-system-x86_64 not found. Install QEMU first."; exit 1; }
	@bash scripts/test-qemu.sh

## docker-build: Build the Docker builder image
docker-build:
	@echo "==> Building Docker builder image..."
	docker build -f Dockerfile.builder -t $(PROJECT)-builder .

## docker-test: Run the build inside Docker
docker-test: docker-build
	@echo "==> Running build inside Docker..."
	docker run --rm \
		-v $(CURDIR)/output:/build/output \
		-e OS_TYPE=$(OS_TYPE) \
		-e OS_BASE=$(OS_BASE) \
		-e OS_HOSTNAME=$(OS_HOSTNAME) \
		$(PROJECT)-builder

## test-services: Run service smoke tests
test-services:
	@bash tests/test-services.sh

## test-network: Run network smoke tests
test-network:
	@bash tests/test-network.sh

## test-all: Run all tests
test-all: test-services test-network
	@echo "All tests passed."

## clean: Remove build output
clean:
	@echo "==> Cleaning output..."
	@rm -rf $(OUTPUT_DIR)
	@echo "Clean complete."

## help: Show this help
help:
	@grep -E '^## ' Makefile | sed 's/## /  make /'
```

## Step 6: Generate `scripts/build-alpine.sh`

Write `<project_name>/scripts/build-alpine.sh`:

```bash
#!/bin/bash
# =============================================================================
# build-alpine.sh — Alpine Linux OS builder
# Uses Docker for reproducible builds
# =============================================================================
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
OUTPUT_DIR="${SCRIPT_DIR}/output"
ROOTFS_DIR="${OUTPUT_DIR}/rootfs"
ALPINE_VERSION="${ALPINE_VERSION:-3.19}"
ALPINE_ARCH="${ALPINE_ARCH:-x86_64}"
ALPINE_MIRROR="${ALPINE_MIRROR:-https://dl-cdn.alpinelinux.org/alpine}"
MINIROOTFS_URL="${ALPINE_MIRROR}/v${ALPINE_VERSION}/releases/${ALPINE_ARCH}/alpine-minirootfs-${ALPINE_VERSION}.0-${ALPINE_ARCH}.tar.gz"

PACKAGES_FILE="${SCRIPT_DIR}/config/packages.txt"
OVERLAY_DIR="${SCRIPT_DIR}/overlay"
ANSWERFILE="${SCRIPT_DIR}/config/alpine/answerfile"

RED='\033[0;31m'
GREEN='\033[0;32m'
BLUE='\033[0;34m'
NC='\033[0m'

log()  { echo -e "${BLUE}[alpine-build]${NC} $*"; }
ok()   { echo -e "${GREEN}[OK]${NC} $*"; }
fail() { echo -e "${RED}[FAIL]${NC} $*"; exit 1; }

check_docker() {
    command -v docker &>/dev/null || fail "Docker is required for Alpine builds."
    docker info &>/dev/null || fail "Docker daemon is not running."
    ok "Docker available"
}

download_minirootfs() {
    local tarball="${OUTPUT_DIR}/alpine-minirootfs.tar.gz"
    if [[ ! -f "$tarball" ]]; then
        log "Downloading Alpine ${ALPINE_VERSION} mini root filesystem..."
        curl -fsSL -o "$tarball" "$MINIROOTFS_URL" \
            || fail "Failed to download Alpine minirootfs from ${MINIROOTFS_URL}"
        ok "Downloaded: $(du -sh "$tarball" | cut -f1)"
    else
        log "Using cached minirootfs: $tarball"
    fi
    echo "$tarball"
}

setup_rootfs() {
    local tarball="$1"
    log "Setting up root filesystem..."
    rm -rf "$ROOTFS_DIR"
    mkdir -p "$ROOTFS_DIR"
    tar -xzf "$tarball" -C "$ROOTFS_DIR"
    ok "Root filesystem extracted to ${ROOTFS_DIR}"
}

configure_apk_repos() {
    log "Configuring APK repositories..."
    cat > "${ROOTFS_DIR}/etc/apk/repositories" <<EOF
${ALPINE_MIRROR}/v${ALPINE_VERSION}/main
${ALPINE_MIRROR}/v${ALPINE_VERSION}/community
EOF
    ok "APK repos configured"
}

install_packages() {
    log "Installing packages..."
    local packages="alpine-base openrc openssh chrony"

    # Read from packages.txt if it exists
    if [[ -f "$PACKAGES_FILE" ]]; then
        while IFS= read -r line; do
            [[ "$line" =~ ^#.*$ || -z "$line" ]] && continue
            packages="${packages} ${line%%#*}"  # strip inline comments
        done < "$PACKAGES_FILE"
    fi

    log "Package list: ${packages}"

    docker run --rm \
        --privileged \
        -v "${ROOTFS_DIR}:/rootfs" \
        alpine:${ALPINE_VERSION} \
        sh -c "
            apk add --no-cache --root /rootfs \
                --repositories-file /rootfs/etc/apk/repositories \
                --initdb \
                ${packages}
        " || fail "Package installation failed"

    ok "Packages installed"
}

apply_overlay() {
    if [[ -d "$OVERLAY_DIR" ]]; then
        log "Applying overlay to rootfs..."
        cp -r "${OVERLAY_DIR}/." "${ROOTFS_DIR}/"
        # Fix permissions on executables
        find "${ROOTFS_DIR}/usr/local/bin" -type f -exec chmod 755 {} \; 2>/dev/null || true
        ok "Overlay applied"
    fi
}

configure_services() {
    log "Configuring services..."
    docker run --rm \
        --privileged \
        -v "${ROOTFS_DIR}:/rootfs" \
        alpine:${ALPINE_VERSION} \
        sh -c "
            # Enable core services
            chroot /rootfs rc-update add devfs sysinit
            chroot /rootfs rc-update add dmesg sysinit
            chroot /rootfs rc-update add mdev sysinit
            chroot /rootfs rc-update add hwclock boot
            chroot /rootfs rc-update add modules boot
            chroot /rootfs rc-update add sysctl boot
            chroot /rootfs rc-update add hostname boot
            chroot /rootfs rc-update add bootmisc boot
            chroot /rootfs rc-update add networking boot
            chroot /rootfs rc-update add sshd default
            chroot /rootfs rc-update add chronyd default
            chroot /rootfs rc-update add crond default
            chroot /rootfs rc-update add mount-ro shutdown
            chroot /rootfs rc-update add killprocs shutdown
            chroot /rootfs rc-update add savecache shutdown
        " || fail "Service configuration failed"
    ok "Services configured"
}

apply_network_config() {
    log "Applying network configuration..."
    cp "${SCRIPT_DIR}/config/network/interfaces" "${ROOTFS_DIR}/etc/network/interfaces" 2>/dev/null || \
    cat > "${ROOTFS_DIR}/etc/network/interfaces" <<'EOF'
auto lo
iface lo inet loopback

auto eth0
iface eth0 inet dhcp
    hostname custom-os
EOF
    cp "${SCRIPT_DIR}/config/network/sshd_config" "${ROOTFS_DIR}/etc/ssh/sshd_config" 2>/dev/null || true
    ok "Network configuration applied"
}

create_disk_image() {
    log "Creating disk image..."
    local image="${OUTPUT_DIR}/os.img"
    local size_mb=512

    dd if=/dev/zero of="$image" bs=1M count=${size_mb} 2>/dev/null
    # Create ext4 filesystem (requires mkfs.ext4, may need to run in Docker)
    docker run --rm \
        --privileged \
        -v "${OUTPUT_DIR}:/output" \
        -v "${ROOTFS_DIR}:/rootfs:ro" \
        alpine:${ALPINE_VERSION} \
        sh -c "
            apk add --no-cache e2fsprogs dosfstools syslinux >/dev/null 2>&1
            mkfs.ext4 -F -L 'customos-root' /output/os.img
            mkdir -p /mnt/target
            mount /output/os.img /mnt/target
            cp -a /rootfs/. /mnt/target/
            umount /mnt/target
        " || {
        warn "Disk image creation requires Docker --privileged. Skipping image creation."
        warn "Root filesystem is available at: ${ROOTFS_DIR}"
        return 0
    }

    ok "Disk image created: ${image} (${size_mb}MB)"
}

main() {
    log "Starting Alpine ${ALPINE_VERSION} build for ${ALPINE_ARCH}..."
    mkdir -p "$OUTPUT_DIR"
    check_docker
    local tarball
    tarball=$(download_minirootfs)
    setup_rootfs "$tarball"
    configure_apk_repos
    install_packages
    apply_overlay
    configure_services
    apply_network_config
    create_disk_image
    ok "Alpine build complete. Rootfs: ${ROOTFS_DIR}"
}

main "$@"
```

## Step 7: Generate `scripts/build-buildroot.sh`

Write `<project_name>/scripts/build-buildroot.sh`:

```bash
#!/bin/bash
# =============================================================================
# build-buildroot.sh — Buildroot embedded OS builder
# =============================================================================
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
OUTPUT_DIR="${SCRIPT_DIR}/output"
BUILDROOT_VERSION="${BUILDROOT_VERSION:-2024.02}"
BUILDROOT_URL="https://buildroot.org/downloads/buildroot-${BUILDROOT_VERSION}.tar.gz"
BUILDROOT_DIR="${OUTPUT_DIR}/buildroot-${BUILDROOT_VERSION}"
DEFCONFIG="${SCRIPT_DIR}/config/buildroot/.config"
OVERLAY_DIR="${SCRIPT_DIR}/overlay"

log()  { echo "[buildroot-build] $*"; }
ok()   { echo "[OK] $*"; }
fail() { echo "[FAIL] $*"; exit 1; }

check_deps() {
    local missing=()
    for cmd in make gcc g++ patch cpio python3 rsync bc; do
        command -v "$cmd" &>/dev/null || missing+=("$cmd")
    done
    if [[ ${#missing[@]} -gt 0 ]]; then
        fail "Missing required tools for Buildroot: ${missing[*]}"
    fi
    ok "Build dependencies satisfied"
}

download_buildroot() {
    if [[ ! -d "$BUILDROOT_DIR" ]]; then
        local tarball="${OUTPUT_DIR}/buildroot-${BUILDROOT_VERSION}.tar.gz"
        log "Downloading Buildroot ${BUILDROOT_VERSION}..."
        mkdir -p "$OUTPUT_DIR"
        curl -fsSL -o "$tarball" "$BUILDROOT_URL" \
            || fail "Failed to download Buildroot"
        log "Extracting Buildroot..."
        tar -xzf "$tarball" -C "$OUTPUT_DIR"
        ok "Buildroot ${BUILDROOT_VERSION} ready"
    else
        log "Using existing Buildroot: ${BUILDROOT_DIR}"
    fi
}

apply_defconfig() {
    log "Applying custom defconfig..."
    if [[ -f "$DEFCONFIG" ]]; then
        cp "$DEFCONFIG" "${BUILDROOT_DIR}/.config"
        make -C "$BUILDROOT_DIR" olddefconfig
        ok "Defconfig applied"
    else
        log "No custom defconfig found — using default x86_64 defconfig"
        make -C "$BUILDROOT_DIR" pc_x86_64_defconfig
    fi
}

configure_overlay() {
    if [[ -d "$OVERLAY_DIR" ]]; then
        log "Configuring Buildroot overlay..."
        # Tell Buildroot to use our overlay directory
        echo "BR2_ROOTFS_OVERLAY=\"${OVERLAY_DIR}\"" >> "${BUILDROOT_DIR}/.config"
        make -C "$BUILDROOT_DIR" olddefconfig
        ok "Overlay configured: ${OVERLAY_DIR}"
    fi
}

run_build() {
    log "Starting Buildroot compilation (this takes 30-90 minutes)..."
    log "Using $(nproc) parallel jobs"
    make -C "$BUILDROOT_DIR" -j"$(nproc)" 2>&1 | tee "${OUTPUT_DIR}/buildroot-build.log"
    ok "Buildroot build complete"
}

collect_output() {
    local images_dir="${BUILDROOT_DIR}/output/images"
    log "Collecting build outputs..."
    mkdir -p "${OUTPUT_DIR}/images"
    if [[ -f "${images_dir}/bzImage" ]]; then
        cp "${images_dir}/bzImage" "${OUTPUT_DIR}/images/"
        ok "Kernel: ${OUTPUT_DIR}/images/bzImage"
    fi
    for img in rootfs.ext4 rootfs.squashfs rootfs.tar; do
        if [[ -f "${images_dir}/${img}" ]]; then
            cp "${images_dir}/${img}" "${OUTPUT_DIR}/images/"
            ok "Image: ${OUTPUT_DIR}/images/${img}"
        fi
    done
    if [[ -f "${images_dir}/disk.img" ]]; then
        cp "${images_dir}/disk.img" "${OUTPUT_DIR}/os.img"
        ok "Disk image: ${OUTPUT_DIR}/os.img"
    fi
}

main() {
    log "Starting Buildroot embedded OS build..."
    mkdir -p "$OUTPUT_DIR"
    check_deps
    download_buildroot
    apply_defconfig
    configure_overlay
    run_build
    collect_output
    ok "Embedded OS build complete. Output: ${OUTPUT_DIR}/images/"
}

main "$@"
```

## Step 8: Generate `scripts/build-debian.sh`

Write `<project_name>/scripts/build-debian.sh`:

```bash
#!/bin/bash
# =============================================================================
# build-debian.sh — Debian-based desktop OS builder using debootstrap
# =============================================================================
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
OUTPUT_DIR="${SCRIPT_DIR}/output"
ROOTFS_DIR="${OUTPUT_DIR}/rootfs"
DEBIAN_RELEASE="${DEBIAN_RELEASE:-bookworm}"
DEBIAN_MIRROR="${DEBIAN_MIRROR:-http://deb.debian.org/debian}"
OVERLAY_DIR="${SCRIPT_DIR}/overlay"
PACKAGES_FILE="${SCRIPT_DIR}/config/packages.txt"

log()  { echo "[debian-build] $*"; }
ok()   { echo "[OK] $*"; }
fail() { echo "[FAIL] $*"; exit 1; }

check_deps() {
    command -v debootstrap &>/dev/null || \
        fail "debootstrap required. Install: apt install debootstrap"
    command -v chroot &>/dev/null || fail "chroot required"
    [[ $EUID -eq 0 ]] || fail "Debian build requires root (sudo ./scripts/build-debian.sh)"
    ok "Dependencies satisfied"
}

run_debootstrap() {
    log "Running debootstrap for Debian ${DEBIAN_RELEASE}..."
    rm -rf "$ROOTFS_DIR"
    mkdir -p "$ROOTFS_DIR"
    debootstrap \
        --arch=amd64 \
        --include=systemd,systemd-sysv,openssh-server,sudo,curl,wget,vim,htop,net-tools,locales \
        "${DEBIAN_RELEASE}" \
        "$ROOTFS_DIR" \
        "$DEBIAN_MIRROR" \
        || fail "debootstrap failed"
    ok "Debootstrap complete"
}

configure_chroot() {
    log "Configuring Debian system..."

    # Bind mounts for chroot
    mount --bind /dev  "${ROOTFS_DIR}/dev"
    mount --bind /proc "${ROOTFS_DIR}/proc"
    mount --bind /sys  "${ROOTFS_DIR}/sys"

    # Add extra packages
    local extra_pkgs=""
    if [[ -f "$PACKAGES_FILE" ]]; then
        extra_pkgs=$(grep -v '^#' "$PACKAGES_FILE" | tr '\n' ' ')
    fi

    chroot "$ROOTFS_DIR" /bin/bash <<CHROOT_EOF
set -e
export DEBIAN_FRONTEND=noninteractive
locale-gen en_US.UTF-8
update-locale LANG=en_US.UTF-8

# Install extra packages
if [[ -n "${extra_pkgs}" ]]; then
    apt-get update -q
    apt-get install -y --no-install-recommends ${extra_pkgs}
fi

# Enable systemd services
systemctl enable ssh 2>/dev/null || true
systemctl enable cron 2>/dev/null || true

# Clean up
apt-get clean
rm -rf /var/lib/apt/lists/*
CHROOT_EOF

    # Unmount bind mounts
    umount "${ROOTFS_DIR}/dev"  || true
    umount "${ROOTFS_DIR}/proc" || true
    umount "${ROOTFS_DIR}/sys"  || true

    ok "Chroot configuration complete"
}

apply_overlay() {
    if [[ -d "$OVERLAY_DIR" ]]; then
        log "Applying overlay..."
        cp -r "${OVERLAY_DIR}/." "${ROOTFS_DIR}/"
        find "${ROOTFS_DIR}/usr/local/bin" -type f -exec chmod 755 {} \;
        ok "Overlay applied"
    fi
}

create_image() {
    log "Creating Debian disk image..."
    local image="${OUTPUT_DIR}/os.img"
    local size_mb=2048
    dd if=/dev/zero of="$image" bs=1M count=${size_mb} status=progress
    mkfs.ext4 -F -L "debian-root" "$image"
    local mnt="${OUTPUT_DIR}/mnt"
    mkdir -p "$mnt"
    mount "$image" "$mnt"
    cp -a "${ROOTFS_DIR}/." "$mnt/"
    umount "$mnt"
    rmdir "$mnt"
    ok "Debian disk image: ${image} (${size_mb}MB)"
}

main() {
    log "Building Debian ${DEBIAN_RELEASE} desktop OS..."
    mkdir -p "$OUTPUT_DIR"
    check_deps
    run_debootstrap
    configure_chroot
    apply_overlay
    create_image
    ok "Debian build complete: ${OUTPUT_DIR}/os.img"
}

main "$@"
```

```bash
#!/bin/bash
# =============================================================================
# firewall.sh — nftables firewall rules
# Generated by OneCommand os-builder
# =============================================================================
set -euo pipefail

command -v nft &>/dev/null || { echo "nftables not found. Trying iptables fallback..."; USE_IPTABLES=1; }

apply_nftables() {
    nft flush ruleset

    nft add table inet filter

    # INPUT chain
    nft add chain inet filter input '{ type filter hook input priority 0; policy drop; }'

    # Allow loopback
    nft add rule inet filter input iifname "lo" accept

    # Allow established/related
    nft add rule inet filter input ct state established,related accept

    # Allow ICMP (ping)
    nft add rule inet filter input ip protocol icmp accept
    nft add rule inet filter input ip6 nexthdr icmpv6 accept

    # Allow SSH
    nft add rule inet filter input tcp dport 22 ct state new limit rate 5/minute accept

    # Allow HTTP and HTTPS (if nginx/web feature enabled)
    nft add rule inet filter input tcp dport { 80, 443 } accept

    # Allow custom application ports (add more as needed)
    # PostgreSQL — only from localhost
    # nft add rule inet filter input ip saddr 127.0.0.1 tcp dport 5432 accept

    # Drop invalid packets
    nft add rule inet filter input ct state invalid drop

    # Log and drop everything else
    nft add rule inet filter input log prefix "DROPPED: " flags all
    nft add rule inet filter input drop

    # FORWARD chain — disabled by default
    nft add chain inet filter forward '{ type filter hook forward priority 0; policy drop; }'

    # OUTPUT chain — allow all outbound
    nft add chain inet filter output '{ type filter hook output priority 0; policy accept; }'

    # Save rules
    nft list ruleset > /etc/nftables.conf 2>/dev/null || \
        nft list ruleset > /etc/nftables/nftables.conf 2>/dev/null || true

    echo "nftables rules applied successfully"
    nft list ruleset
}

apply_iptables_fallback() {
    local IPT="iptables"
    local IPT6="ip6tables"

    # Flush existing rules
    $IPT -F && $IPT -X && $IPT -Z
    $IPT -t nat -F && $IPT -t mangle -F

    # Default policies
    $IPT -P INPUT DROP
    $IPT -P FORWARD DROP
    $IPT -P OUTPUT ACCEPT

    # Allow loopback
    $IPT -A INPUT -i lo -j ACCEPT
    $IPT -A OUTPUT -o lo -j ACCEPT

    # Allow established/related
    $IPT -A INPUT -m conntrack --ctstate ESTABLISHED,RELATED -j ACCEPT

    # Allow ICMP
    $IPT -A INPUT -p icmp --icmp-type echo-request -m limit --limit 5/s -j ACCEPT

    # Allow SSH with rate limiting
    $IPT -A INPUT -p tcp --dport 22 -m conntrack --ctstate NEW -m recent --set --name SSH --rsource
    $IPT -A INPUT -p tcp --dport 22 -m conntrack --ctstate NEW -m recent --update --seconds 60 --hitcount 5 --name SSH --rsource -j DROP
    $IPT -A INPUT -p tcp --dport 22 -j ACCEPT

    # Allow HTTP/HTTPS
    $IPT -A INPUT -p tcp --dport 80  -j ACCEPT
    $IPT -A INPUT -p tcp --dport 443 -j ACCEPT

    # Log drops
    $IPT -A INPUT -j LOG --log-prefix "IPTABLES-DROP: " --log-level 4
    $IPT -A INPUT -j DROP

    # IPv6 — same rules
    $IPT6 -F && $IPT6 -X
    $IPT6 -P INPUT DROP
    $IPT6 -P FORWARD DROP
    $IPT6 -P OUTPUT ACCEPT
    $IPT6 -A INPUT -i lo -j ACCEPT
    $IPT6 -A INPUT -m conntrack --ctstate ESTABLISHED,RELATED -j ACCEPT
    $IPT6 -A INPUT -p ipv6-icmp -j ACCEPT
    $IPT6 -A INPUT -p tcp --dport 22  -j ACCEPT
    $IPT6 -A INPUT -p tcp --dport 80  -j ACCEPT
    $IPT6 -A INPUT -p tcp --dport 443 -j ACCEPT
    $IPT6 -A INPUT -j DROP

    echo "iptables rules applied"
}

if [[ "${USE_IPTABLES:-0}" == "1" ]]; then
    apply_iptables_fallback
else
    apply_nftables
fi
```

## Step 16: Generate `scripts/build-iso.sh`

```bash
#!/bin/bash
# =============================================================================
# build-iso.sh — Create bootable ISO from rootfs using xorriso + GRUB2
# =============================================================================
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
OUTPUT_DIR="${SCRIPT_DIR}/output"
ROOTFS_DIR="${OUTPUT_DIR}/rootfs"
ISO_STAGING="${OUTPUT_DIR}/iso-staging"
ISO_OUTPUT="${OUTPUT_DIR}/os.iso"
OS_NAME="${OS_NAME:-CustomOS}"
GRUB_CFG="${SCRIPT_DIR}/overlay/boot/grub/grub.cfg"

log()  { echo "[build-iso] $*"; }
ok()   { echo "[OK] $*"; }
fail() { echo "[FAIL] $*"; exit 1; }

check_deps() {
    command -v xorriso &>/dev/null || fail "xorriso required. Install: apk add xorriso"
    command -v grub-mkimage &>/dev/null || \
    command -v grub2-mkimage &>/dev/null || \
        fail "grub-mkimage required. Install: apk add grub"
    ok "ISO build tools available"
}

prepare_staging() {
    log "Preparing ISO staging directory..."
    rm -rf "$ISO_STAGING"
    mkdir -p "${ISO_STAGING}/boot/grub"
    mkdir -p "${ISO_STAGING}/isolinux"
    mkdir -p "${ISO_STAGING}/EFI/BOOT"
    ok "Staging directory ready: ${ISO_STAGING}"
}

install_grub() {
    log "Installing GRUB2 bootloader..."

    # Copy kernel and initrd from rootfs
    local kernel initrd
    kernel=$(find "${ROOTFS_DIR}/boot" -name 'vmlinuz*' | sort | tail -1 2>/dev/null) || kernel=""
    initrd=$(find "${ROOTFS_DIR}/boot" -name 'initrd*' -o -name 'initramfs*' | sort | tail -1 2>/dev/null) || initrd=""

    if [[ -n "$kernel" ]]; then
        cp "$kernel" "${ISO_STAGING}/boot/vmlinuz"
        ok "Kernel: $kernel"
    else
        warn() { echo "[WARN] $*"; }
        warn "No kernel found in rootfs/boot — ISO may not boot. Run build.sh first."
    fi

    if [[ -n "$initrd" ]]; then
        cp "$initrd" "${ISO_STAGING}/boot/initrd"
        ok "Initrd: $initrd"
    fi

    # Generate GRUB configuration
    cat > "${ISO_STAGING}/boot/grub/grub.cfg" <<GRUBCFG
# =============================================================================
# GRUB2 Boot Configuration — ${OS_NAME}
# =============================================================================

set default=0
set timeout=5
set timeout_style=menu

insmod all_video
insmod gfxterm
terminal_output gfxterm

if background_image /boot/grub/splash.png; then
    set color_normal=white/black
    set color_highlight=black/white
fi

menuentry "${OS_NAME} (default)" {
    linux /boot/vmlinuz root=/dev/sda1 rw quiet console=tty0 console=ttyS0,115200n8 apparmor=1 security=apparmor
    initrd /boot/initrd
}

menuentry "${OS_NAME} (verbose)" {
    linux /boot/vmlinuz root=/dev/sda1 rw loglevel=7 console=tty0 console=ttyS0,115200n8
    initrd /boot/initrd
}

menuentry "${OS_NAME} (rescue — single user)" {
    linux /boot/vmlinuz root=/dev/sda1 rw single init=/bin/sh console=ttyS0,115200n8
    initrd /boot/initrd
}

menuentry "Memory Test (memtest86+)" {
    linux16 /boot/memtest86+
}

menuentry "Boot from disk" {
    chainloader (hd0)+1
}
GRUBCFG

    ok "GRUB configuration written"
}

pack_rootfs_to_squashfs() {
    log "Packing rootfs into SquashFS..."
    if command -v mksquashfs &>/dev/null; then
        mksquashfs "${ROOTFS_DIR}" "${ISO_STAGING}/rootfs.squashfs" \
            -comp xz \
            -Xdict-size 100% \
            -noappend \
            -e proc sys dev tmp \
            2>&1 | tail -5
        ok "SquashFS: $(du -sh "${ISO_STAGING}/rootfs.squashfs" | cut -f1)"
    else
        log "mksquashfs not available — skipping rootfs packing (use Docker builder)"
    fi
}

create_iso() {
    log "Creating bootable ISO with xorriso..."

    # Determine GRUB binary path (varies by distro)
    local grub_core=""
    for candidate in \
        "/usr/lib/grub/i386-pc/core.img" \
        "/usr/lib/grub2/i386-pc/core.img" \
        "/usr/lib64/grub/i386-pc/core.img"; do
        [[ -f "$candidate" ]] && grub_core="$candidate" && break
    done

    xorriso -as mkisofs \
        -iso-level 3 \
        -full-iso9660-filenames \
        -volid "${OS_NAME}" \
        -appid "${OS_NAME} ISO" \
        -publisher "OneCommand OS Builder" \
        -preparer "OneCommand" \
        -output "${ISO_OUTPUT}" \
        -eltorito-boot isolinux/isolinux.bin \
        -eltorito-catalog isolinux/boot.cat \
        -no-emul-boot \
        -boot-load-size 4 \
        -boot-info-table \
        --grub2-boot-info \
        --grub2-mbr /usr/lib/grub/i386-pc/boot_hybrid.img \
        -eltorito-alt-boot \
        -e EFI/BOOT/efiboot.img \
        -no-emul-boot \
        -isohybrid-gpt-basdat \
        "${ISO_STAGING}" 2>&1 | tail -20

    ok "ISO created: ${ISO_OUTPUT} ($(du -sh "$ISO_OUTPUT" | cut -f1))"
}

main() {
    log "Building bootable ISO for ${OS_NAME}..."
    [[ -d "$ROOTFS_DIR" ]] || fail "Rootfs not found at ${ROOTFS_DIR}. Run build.sh first."
    mkdir -p "$OUTPUT_DIR"
    check_deps
    prepare_staging
    install_grub
    pack_rootfs_to_squashfs
    create_iso
    echo ""
    echo "ISO build complete!"
    echo "  ISO: ${ISO_OUTPUT}"
    echo "  Burn to USB: dd if=${ISO_OUTPUT} of=/dev/sdX bs=4M status=progress"
    echo "  Test: make test  (boots in QEMU)"
}

main "$@"
```

```bash
#!/bin/bash
# =============================================================================
# os-status — System health overview
# =============================================================================

RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
CYAN='\033[0;36m'
BOLD='\033[1m'
NC='\033[0m'

bar() {
    local pct=$1 width=30
    local filled=$(( pct * width / 100 ))
    local empty=$(( width - filled ))
    local color=$GREEN
    (( pct > 70 )) && color=$YELLOW
    (( pct > 90 )) && color=$RED
    printf "${color}"
    printf '%0.s█' $(seq 1 $filled)
    printf "${NC}"
    printf '%0.s░' $(seq 1 $empty)
    printf " %d%%" "$pct"
}

print_header() {
    echo -e "${BOLD}${BLUE}"
    echo "╔══════════════════════════════════════════════════════╗"
    echo "║              SYSTEM STATUS — $(hostname -s)          "
    echo "╚══════════════════════════════════════════════════════╝"
    echo -e "${NC}"
}

print_cpu() {
    # Load averages from /proc/loadavg
    read -r load1 load5 load15 _ _ < /proc/loadavg
    local ncpu
    ncpu=$(nproc 2>/dev/null || grep -c ^processor /proc/cpuinfo)
    local pct
    pct=$(awk "BEGIN { printf \"%d\", ($load1 / $ncpu) * 100 }")
    (( pct > 100 )) && pct=100

    echo -e "${BOLD}CPU${NC}"
    printf "  Load (1/5/15m) : ${CYAN}%s  %s  %s${NC}\n" "$load1" "$load5" "$load15"
    printf "  Utilization    : "
    bar "$pct"
    echo ""
}

print_memory() {
    local total_kb avail_kb used_kb used_pct
    total_kb=$(awk '/^MemTotal/{print $2}' /proc/meminfo)
    avail_kb=$(awk '/^MemAvailable/{print $2}' /proc/meminfo)
    used_kb=$(( total_kb - avail_kb ))
    used_pct=$(( used_kb * 100 / total_kb ))

    local total_mb=$(( total_kb / 1024 ))
    local used_mb=$(( used_kb / 1024 ))
    local avail_mb=$(( avail_kb / 1024 ))

    echo -e "${BOLD}MEMORY${NC}"
    printf "  Total / Used   : ${CYAN}%dMB / %dMB${NC} (%dMB free)\n" "$total_mb" "$used_mb" "$avail_mb"
    printf "  Utilization    : "
    bar "$used_pct"
    echo ""
}

print_disk() {
    echo -e "${BOLD}DISK${NC}"
    df -h --output=target,size,used,avail,pcent 2>/dev/null | \
    grep -E '^(/|/home|/var|/tmp)' | while read -r mp size used avail pct; do
        local p="${pct//%/}"
        printf "  %-12s : ${CYAN}%s used / %s total${NC} (%s avail)\n" \
            "$mp" "$used" "$size" "$avail"
        printf "               "
        bar "$p"
        echo ""
    done
}

print_network() {
    echo -e "${BOLD}NETWORK${NC}"
    ip -brief addr show 2>/dev/null | grep -v '^lo' | while read -r iface state addrs; do
        printf "  %-12s : ${CYAN}%-20s${NC} [%s]\n" "$iface" "${addrs:-no IP}" "$state"
    done
    echo ""
}

print_services() {
    echo -e "${BOLD}SERVICES${NC}"
    local services=(sshd nginx postgresql docker chronyd crond)
    for svc in "${services[@]}"; do
        # Try systemctl first, then rc-service (OpenRC)
        local status color
        if command -v systemctl &>/dev/null; then
            if systemctl is-active --quiet "$svc" 2>/dev/null; then
                status="running"; color=$GREEN
            else
                # Check if unit exists
                if systemctl list-unit-files "${svc}.service" &>/dev/null; then
                    status="stopped"; color=$RED
                else
                    continue  # service not installed
                fi
            fi
        elif command -v rc-service &>/dev/null; then
            if rc-service "$svc" status &>/dev/null; then
                status="running"; color=$GREEN
            else
                status="stopped"; color=$RED
            fi
        else
            if pgrep -x "$svc" &>/dev/null; then
                status="running"; color=$GREEN
            else
                status="unknown"; color=$YELLOW
            fi
        fi
        printf "  %-16s : ${color}%s${NC}\n" "$svc" "$status"
    done
    echo ""
}

print_uptime() {
    echo -e "${BOLD}UPTIME${NC}"
    local uptime_s
    read -r uptime_s _ < /proc/uptime
    local days=$(( ${uptime_s%.*} / 86400 ))
    local hours=$(( (${uptime_s%.*} % 86400) / 3600 ))
    local mins=$(( (${uptime_s%.*} % 3600) / 60 ))
    printf "  Up: ${CYAN}%dd %dh %dm${NC}  |  " "$days" "$hours" "$mins"
    printf "Date: ${CYAN}%s${NC}\n\n" "$(date '+%Y-%m-%d %H:%M:%S %Z')"
}

print_header
print_uptime
print_cpu
print_memory
print_disk
print_network
print_services
```

```bash
#!/bin/bash
# =============================================================================
# os-update — Safe OS update with rollback support
# =============================================================================
set -euo pipefail

[[ $EUID -ne 0 ]] && { echo "Run as root: sudo os-update"; exit 1; }

SNAPSHOT_DIR="/var/backups/os-snapshots"
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
LOG="/var/log/os-update-${TIMESTAMP}.log"

log()  { echo "[os-update] $*" | tee -a "$LOG"; }
ok()   { echo "[OK] $*" | tee -a "$LOG"; }
fail() { echo "[FAIL] $*" | tee -a "$LOG"; exit 1; }

detect_package_manager() {
    if command -v apk &>/dev/null; then echo "apk"
    elif command -v apt-get &>/dev/null; then echo "apt"
    elif command -v dnf &>/dev/null; then echo "dnf"
    elif command -v yum &>/dev/null; then echo "yum"
    else fail "No supported package manager found"; fi
}

snapshot_config() {
    log "Creating config snapshot before update..."
    mkdir -p "$SNAPSHOT_DIR"
    local snap="${SNAPSHOT_DIR}/pre-update-${TIMESTAMP}.tar.gz"
    tar -czf "$snap" \
        /etc/ssh/ \
        /etc/network/ \
        /etc/hostname \
        /etc/hosts \
        /usr/local/bin/os-* \
        2>/dev/null || true
    ok "Snapshot: $snap"
}

do_update() {
    local pm
    pm=$(detect_package_manager)
    log "Updating packages with ${pm}..."
    case "$pm" in
        apk)
            apk update 2>&1 | tee -a "$LOG"
            apk upgrade 2>&1 | tee -a "$LOG"
            ;;
        apt)
            DEBIAN_FRONTEND=noninteractive apt-get update -q 2>&1 | tee -a "$LOG"
            DEBIAN_FRONTEND=noninteractive apt-get upgrade -y -q 2>&1 | tee -a "$LOG"
            apt-get autoremove -y -q 2>&1 | tee -a "$LOG"
            ;;
        dnf|yum)
            $pm update -y 2>&1 | tee -a "$LOG"
            ;;
    esac
    ok "Package update complete"
}

verify_services() {
    log "Verifying critical services after update..."
    local failed=0
    for svc in sshd networking; do
        if command -v systemctl &>/dev/null; then
            systemctl is-active --quiet "$svc" 2>/dev/null || {
                log "WARNING: ${svc} not running after update"
                systemctl start "$svc" 2>/dev/null || true
                ((failed++)) || true
            }
        fi
    done
    [[ $failed -eq 0 ]] && ok "All critical services running" || \
        log "WARNING: $failed service(s) had issues — check $LOG"
}

list_snapshots() {
    echo "Available snapshots for rollback:"
    ls -lh "${SNAPSHOT_DIR}"/*.tar.gz 2>/dev/null || echo "  No snapshots found"
}

rollback() {
    local snap="$1"
    [[ -f "$snap" ]] || { list_snapshots; fail "Specify snapshot file: os-update --rollback /path/to/snap.tar.gz"; }
    log "Rolling back from: $snap"
    tar -xzf "$snap" -C / 2>&1 | tee -a "$LOG"
    ok "Rollback complete"
}

case "${1:-update}" in
    --rollback) rollback "${2:-}" ;;
    --list)     list_snapshots ;;
    update|"")
        snapshot_config
        do_update
        verify_services
        ok "Update complete. Log: $LOG"
        ;;
    *) echo "Usage: os-update [update|--rollback <snapshot>|--list]"; exit 1 ;;
esac
```

```bash
#!/bin/bash
# =============================================================================
# os-backup — Backup critical OS configuration
# =============================================================================
set -euo pipefail

[[ $EUID -ne 0 ]] && { echo "Run as root: sudo os-backup"; exit 1; }

BACKUP_DIR="${BACKUP_DIR:-/var/backups/os-config}"
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
HOSTNAME=$(hostname -s)
BACKUP_FILE="${BACKUP_DIR}/${HOSTNAME}-backup-${TIMESTAMP}.tar.gz"
LATEST_LINK="${BACKUP_DIR}/${HOSTNAME}-latest.tar.gz"
KEEP_BACKUPS="${KEEP_BACKUPS:-7}"

log()  { echo "[os-backup] $*"; }
ok()   { echo "[OK] $*"; }
fail() { echo "[FAIL] $*"; exit 1; }

BACKUP_PATHS=(
    /etc/ssh/
    /etc/network/
    /etc/nftables.conf
    /etc/hostname
    /etc/hosts
    /etc/resolv.conf
    /etc/crontab
    /etc/cron.d/
    /usr/local/bin/os-*
    /etc/motd
    /root/.ssh/
    /home/*/.ssh/
)

create_backup() {
    log "Starting backup to: $BACKUP_FILE"
    mkdir -p "$BACKUP_DIR"

    # Build include list (only existing paths)
    local include_args=()
    for p in "${BACKUP_PATHS[@]}"; do
        [[ -e "$p" || -d "$p" ]] && include_args+=("$p")
    done

    tar -czf "$BACKUP_FILE" \
        --ignore-failed-read \
        "${include_args[@]}" 2>/dev/null

    ln -sf "$BACKUP_FILE" "$LATEST_LINK"
    ok "Backup created: $BACKUP_FILE ($(du -sh "$BACKUP_FILE" | cut -f1))"
}

prune_old_backups() {
    log "Pruning backups older than ${KEEP_BACKUPS} days..."
    find "$BACKUP_DIR" -name "*.tar.gz" -not -newer "$LATEST_LINK" \
        -mtime "+${KEEP_BACKUPS}" -delete 2>/dev/null || true
    local count
    count=$(find "$BACKUP_DIR" -name "*.tar.gz" | wc -l)
    ok "Kept ${count} backup(s)"
}

list_backups() {
    echo "Available backups in ${BACKUP_DIR}:"
    ls -lh "${BACKUP_DIR}"/*.tar.gz 2>/dev/null || echo "  No backups found"
}

restore_backup() {
    local file="${1:-$LATEST_LINK}"
    [[ -f "$file" ]] || fail "Backup not found: $file"
    echo "WARNING: This will overwrite existing config files."
    read -rp "Restore from ${file}? [y/N] " confirm
    [[ "$confirm" =~ ^[Yy]$ ]] || { echo "Aborted."; exit 0; }
    tar -xzf "$file" -C / --overwrite 2>/dev/null
    ok "Restored from: $file"
}

case "${1:-backup}" in
    backup|"") create_backup; prune_old_backups ;;
    --list)    list_backups ;;
    --restore) restore_backup "${2:-}" ;;
    *) echo "Usage: os-backup [backup|--list|--restore [file]]"; exit 1 ;;
esac
```

```bash
#!/bin/bash
# =============================================================================
# os-monitor — Real-time system monitoring (terminal dashboard)
# =============================================================================

INTERVAL="${INTERVAL:-2}"
DAEMON_MODE="${1:-}"

RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
CYAN='\033[0;36m'
BOLD='\033[1m'
NC='\033[0m'
CLEAR='\033[2J\033[H'

sparkline() {
    # Accepts space-separated values, prints Unicode spark bars
    local vals=($*)
    local max=0
    for v in "${vals[@]}"; do
        (( v > max )) && max=$v
    done
    (( max == 0 )) && max=1
    local chars=("▁" "▂" "▃" "▄" "▅" "▆" "▇" "█")
    for v in "${vals[@]}"; do
        local idx=$(( v * 7 / max ))
        (( idx > 7 )) && idx=7
        printf "%s" "${chars[$idx]}"
    done
}

declare -a CPU_HIST=()
declare -a MEM_HIST=()
NET_RX_PREV=0
NET_TX_PREV=0

get_cpu_pct() {
    # Read /proc/stat twice with a short sleep for delta
    local stat1 stat2
    read -r _ u1 n1 s1 i1 _ < /proc/stat
    sleep 0.1
    read -r _ u2 n2 s2 i2 _ < /proc/stat
    local total1=$(( u1+n1+s1+i1 ))
    local total2=$(( u2+n2+s2+i2 ))
    local dtotal=$(( total2 - total1 ))
    local didle=$(( i2 - i1 ))
    (( dtotal == 0 )) && echo 0 && return
    echo $(( (dtotal - didle) * 100 / dtotal ))
}

get_mem_pct() {
    local total avail
    total=$(awk '/^MemTotal/{print $2}' /proc/meminfo)
    avail=$(awk '/^MemAvailable/{print $2}' /proc/meminfo)
    echo $(( (total - avail) * 100 / total ))
}

get_net_stats() {
    # Returns rx_bytes tx_bytes for primary non-loopback interface
    local iface
    iface=$(ip route show default 2>/dev/null | awk '/default/{print $5}' | head -1)
    [[ -z "$iface" ]] && { echo "0 0"; return; }
    local rx tx
    rx=$(cat "/sys/class/net/${iface}/statistics/rx_bytes" 2>/dev/null || echo 0)
    tx=$(cat "/sys/class/net/${iface}/statistics/tx_bytes" 2>/dev/null || echo 0)
    echo "$rx $tx"
}

fmt_bytes() {
    local b=$1
    if   (( b >= 1073741824 )); then printf "%.1fGB" "$(echo "scale=1; $b/1073741824" | bc)"
    elif (( b >= 1048576 ));    then printf "%.1fMB" "$(echo "scale=1; $b/1048576" | bc)"
    elif (( b >= 1024 ));       then printf "%.1fKB" "$(echo "scale=1; $b/1024" | bc)"
    else printf "%dB" "$b"; fi
}

render_dashboard() {
    local cpu_pct mem_pct
    cpu_pct=$(get_cpu_pct)
    mem_pct=$(get_mem_pct)

    CPU_HIST+=("$cpu_pct")
    MEM_HIST+=("$mem_pct")
    (( ${#CPU_HIST[@]} > 40 )) && CPU_HIST=("${CPU_HIST[@]:1}")
    (( ${#MEM_HIST[@]} > 40 )) && MEM_HIST=("${MEM_HIST[@]:1}")

    local net_stats rx tx
    net_stats=$(get_net_stats)
    rx=$(echo "$net_stats" | awk '{print $1}')
    tx=$(echo "$net_stats" | awk '{print $2}')
    local rx_delta=$(( rx - NET_RX_PREV ))
    local tx_delta=$(( tx - NET_TX_PREV ))
    NET_RX_PREV=$rx
    NET_TX_PREV=$tx

    local load1 load5 load15
    read -r load1 load5 load15 _ _ < /proc/loadavg

    # Color logic
    local cpu_color=$GREEN mem_color=$GREEN
    (( cpu_pct > 70 )) && cpu_color=$YELLOW
    (( cpu_pct > 90 )) && cpu_color=$RED
    (( mem_pct > 70 )) && mem_color=$YELLOW
    (( mem_pct > 90 )) && mem_color=$RED

    printf "%b" "$CLEAR"
    echo -e "${BOLD}${CYAN}  OS MONITOR — $(hostname -s) — $(date '+%H:%M:%S')${NC}"
    echo    "  ─────────────────────────────────────────────────────"
    echo -e "  ${BOLD}CPU${NC}  ${cpu_color}${cpu_pct}%${NC}  load: ${load1} ${load5} ${load15}"
    printf  "  spark: %s\n" "$(sparkline "${CPU_HIST[@]}")"
    echo ""
    echo -e "  ${BOLD}MEM${NC}  ${mem_color}${mem_pct}%${NC}"
    printf  "  spark: %s\n" "$(sparkline "${MEM_HIST[@]}")"
    echo ""
    echo -e "  ${BOLD}NET${NC}  ↓ $(fmt_bytes $rx_delta)/s  ↑ $(fmt_bytes $tx_delta)/s"
    echo ""
    echo -e "  ${BOLD}DISK${NC}"
    df -h --output=target,pcent 2>/dev/null | grep -E '^(/|/var|/home)' | \
    while read -r mp pct; do
        printf "  %-12s %s\n" "$mp" "$pct"
    done
    echo ""
    echo    "  ─────────────────────────────────────────────────────"
    echo    "  [Ctrl+C to exit]  interval: ${INTERVAL}s"
}

if [[ "$DAEMON_MODE" == "--daemon" ]]; then
    # Run as systemd service — log to journal
    while true; do
        cpu_pct=$(get_cpu_pct)
        mem_pct=$(get_mem_pct)
        load1=$(awk '{print $1}' /proc/loadavg)
        echo "cpu=${cpu_pct}% mem=${mem_pct}% load=${load1}"
        sleep "$INTERVAL"
    done
else
    # Interactive terminal dashboard
    trap 'echo ""; exit 0' INT TERM
    while true; do
        render_dashboard
        sleep "$INTERVAL"
    done
fi
```

```bash
#!/bin/bash
# =============================================================================
# test-boot.sh — Verify the OS boots in QEMU (smoke test)
# =============================================================================
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
OUTPUT_DIR="${SCRIPT_DIR}/output"
ISO="${OUTPUT_DIR}/os.iso"
TIMEOUT="${TIMEOUT:-60}"

fail() { echo "[FAIL] $*"; exit 1; }
ok()   { echo "[OK] $*"; }

command -v qemu-system-x86_64 &>/dev/null || fail "QEMU not available"
[[ -f "$ISO" ]] || fail "ISO not found at ${ISO}. Run: make iso"

echo "Testing boot with QEMU (timeout: ${TIMEOUT}s)..."

# Boot QEMU and capture serial output
output=$(timeout "$TIMEOUT" qemu-system-x86_64 \
    -machine q35 \
    -m 512M \
    -cdrom "$ISO" \
    -boot d \
    -nographic \
    -serial stdio \
    -no-reboot \
    2>/dev/null || true)

# Check for successful boot indicators
if echo "$output" | grep -qiE "login:|Welcome to|hostname|OpenRC"; then
    ok "Boot test PASSED — OS reached login prompt"
elif echo "$output" | grep -qi "Kernel panic"; then
    fail "Boot test FAILED — Kernel panic detected"
else
    echo "[WARN] Boot test inconclusive — check output manually"
    echo "$output" | tail -20
    exit 1
fi
```

```bash
#!/bin/bash
# =============================================================================
# test-services.sh — Verify expected services are running
# =============================================================================

PASS=0; FAIL=0

check_service() {
    local name="$1"
    if command -v systemctl &>/dev/null; then
        systemctl is-active --quiet "$name" 2>/dev/null && {
            echo "[PASS] Service running: $name"
            ((PASS++)) || true
            return
        }
    fi
    if command -v rc-service &>/dev/null; then
        rc-service "$name" status &>/dev/null && {
            echo "[PASS] Service running: $name"
            ((PASS++)) || true
            return
        }
    fi
    if pgrep -x "$name" &>/dev/null; then
        echo "[PASS] Process found: $name"
        ((PASS++)) || true
        return
    fi
    echo "[SKIP] Service not found (may not be installed): $name"
}

check_port() {
    local port="$1" desc="$2"
    if ss -tulpn 2>/dev/null | grep -q ":${port} "; then
        echo "[PASS] Port listening: ${port} (${desc})"
        ((PASS++)) || true
    else
        echo "[SKIP] Port not listening: ${port} (${desc}) — may not be enabled"
    fi
}

echo "=== Service Tests ==="
check_service sshd
check_service nginx
check_service postgresql
check_service docker
check_service chronyd
check_service crond

echo ""
echo "=== Port Tests ==="
check_port 22   "SSH"
check_port 80   "HTTP"
check_port 443  "HTTPS"
check_port 5432 "PostgreSQL"

echo ""
echo "=== Summary ==="
echo "PASS: $PASS  FAIL: $FAIL"
[[ $FAIL -eq 0 ]] || exit 1
```

### `tests/test-network.sh`

```bash
#!/bin/bash
# =============================================================================
# test-network.sh — Verify networking is functional
# =============================================================================

PASS=0; FAIL=0; SKIP=0

pass() { echo "[PASS] $*"; ((PASS++)) || true; }
fail() { echo "[FAIL] $*"; ((FAIL++)) || true; }
skip() { echo "[SKIP] $*"; ((SKIP++)) || true; }

echo "=== Network Interface Tests ==="

# Check loopback
ip link show lo &>/dev/null && pass "Loopback interface up" || fail "Loopback not found"

# Check at least one non-loopback interface
if ip -brief link show | grep -v '^lo' | grep -qi 'UP\|UNKNOWN'; then
    pass "Non-loopback interface is UP"
else
    fail "No active non-loopback network interface"
fi

# Check IP assignment
if ip -brief addr show | grep -v '^lo' | grep -qE '[0-9]+\.[0-9]+'; then
    IP=$(ip -brief addr show | grep -v '^lo' | grep -oE '[0-9]+\.[0-9]+\.[0-9]+\.[0-9]+' | head -1)
    pass "IP address assigned: $IP"
else
    fail "No IP address assigned to non-loopback interface"
fi

echo ""
echo "=== Connectivity Tests ==="

# DNS resolution
if nslookup google.com &>/dev/null 2>&1 || \
   getent hosts google.com &>/dev/null 2>&1; then
    pass "DNS resolution working"
else
    fail "DNS resolution failed"
fi

# Internet connectivity (ping)
if ping -c 2 -W 3 1.1.1.1 &>/dev/null 2>&1; then
    pass "Internet reachable (ping 1.1.1.1)"
else
    skip "Internet not reachable (might be expected in sandbox)"
fi

# HTTP
if command -v curl &>/dev/null; then
    if curl -sf --max-time 5 http://google.com &>/dev/null; then
        pass "HTTP connectivity working"
    else
        skip "HTTP unreachable (might be expected in sandbox)"
    fi
fi

echo ""
echo "=== Firewall Tests ==="

if command -v nft &>/dev/null; then
    if nft list ruleset 2>/dev/null | grep -q 'inet filter'; then
        pass "nftables ruleset loaded"
    else
        fail "nftables ruleset empty — run config/network/firewall.sh"
    fi
elif command -v iptables &>/dev/null; then
    if iptables -L INPUT 2>/dev/null | grep -q 'DROP\|REJECT'; then
        pass "iptables INPUT policy is restrictive"
    else
        fail "iptables has no DROP/REJECT rules on INPUT"
    fi
else
    skip "No firewall tool found (nft / iptables)"
fi

echo ""
echo "=== Summary ==="
echo "PASS: $PASS  FAIL: $FAIL  SKIP: $SKIP"
[[ $FAIL -eq 0 ]] || exit 1
```
